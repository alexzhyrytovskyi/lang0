`=========================================================================================================
`  Finding maximum between 3 random numbers
`=========================================================================================================

print("Finding maximum between 3 random numbers")
randomize()
for (i = 0, i < 10, i++) {
	a = random(1, 100)
	b = random(1, 100)
	c = random(1, 100)
	m = max(a, b, c)
	print("max({a}, {b}, {c}) = {m}")
}
