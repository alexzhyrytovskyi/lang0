print("Знаходження максимального з 3-х випадкових чисел")
randomize()
for (i = 0, i < 10, i++) {
	a = random(1, 100)
	b = random(1, 100)
	c = random(1, 100)
	m = max(a, b, c)
	print("max({a}, {b}, {c}) = {m}")
}
